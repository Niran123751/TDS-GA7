Sample Code: KPI Calculation (Python)


import math

# Compute EPS, NPV and simple YoY growth
net_income = 1.02e9
shares_out = 4.83e8

eps = net_income / shares_out
cashflows = [-2.0e9, 1.2e9, 1.1e9, 0.9e9]
r = 0.09
npv = sum(cf / ((1 + r) ** t) for t, cf in enumerate(cashflows))

def yoy(curr, prev):
    return (curr - prev) / prev

print(f"EPS: {eps:.2f}")
print(f"NPV: ${npv:,.0f}")
print(f"YoY: {yoy(5.4, 5.0):.1%}")
